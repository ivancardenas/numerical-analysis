          \State $p_{i1} \gets points_{i * 2 + 1}$
        \EndFor

          \For {$i = j - 1$ \textbf{to} $n - 1$}
            \State $p_{ij} \gets (p_{(i - 1)(j - 1)} - p_{(i)(j - 1)}) /
            (p_{(i - j + 1)(0)} - p_{i0})$
          \EndFor
        \EndFor

      \EndProcedure
